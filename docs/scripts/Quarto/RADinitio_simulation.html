1 Population Simulation

#!/bin/bash -l

#SBATCH --job-name=rad_pop   
#SBATCH --nodes=1
#SBATCH --ntasks=1
#SBATCH --cpus-per-task=8
#SBATCH --mem=10G
#SBATCH --time=120:00:00
#SBATCH --account=ag-demeaux
#SBATCH --mail-type=END,FAIL
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/rad_pop_%j.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/rad_pop_%j.err

# Create directories
mkdir -p /scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs

# Thread control
export OMP_NUM_THREADS=$SLURM_CPUS_PER_TASK
export OPENBLAS_NUM_THREADS=$SLURM_CPUS_PER_TASK
export MKL_NUM_THREADS=$SLURM_CPUS_PER_TASK

echo "=== Starting Population Simulation at $(date) ==="

mkdir -p /scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/

# Run population simulation only - FIXED genome path
radinitio --make-population \
   --genome /scratch/PNeuhaus/PaulaNeuhaus/Output/mapping/ref_genomes/HM/hypericumpulchrum/hypericumpulchrum_SAMEA10241452.fasta \
   --out-dir /scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/ \
   --n-pops 10 \
   --n-seq-indv 5

if [ $? -eq 0 ]; then
   echo "Population simulation completed successfully at $(date)"

   # Check outputs
   echo "=== Output files ==="
   ls -lh /scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/

   # Check for VCF file
   VCF_FILE="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/ref_loci_vars/ri_master.vcf.gz"
   if [ -f "$VCF_FILE" ]; then
       echo "✓ VCF file created: $VCF_FILE"
       echo "  Number of variants: $(zcat $VCF_FILE 2>/dev/null | grep -v '^#' | wc -l || echo 'unable to count')"
   else
       echo "✗ WARNING: VCF file not found!"
       echo "  Contents of population directory:"
       find /scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/ -type f -name "*.vcf*" 2>/dev/null || echo "  No VCF files found"
   fi

   # Check popmap
   POPMAP_FILE="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/popmap.tsv"
   if [ -f "$POPMAP_FILE" ]; then
       echo "✓ Popmap file created: $POPMAP_FILE"
       echo "  Number of individuals: $(wc -l < $POPMAP_FILE)"
       echo "  First 5 lines:"
       head -5 "$POPMAP_FILE"
   else
       echo "✗ WARNING: Popmap file not found!"
   fi
else
   echo "Population simulation failed at $(date)" >&2
   exit 1
fi

echo "=== Population Simulation Complete ==="

#!/bin/bash -l

#SBATCH --job-name=map_radinitio_KpnI
#SBATCH --nodes=1
#SBATCH --time=1:00:00
#SBATCH --cpus-per-task=6
#SBATCH --mem=32gb
#SBATCH --array=1-50
#SBATCH --account=ag-demeaux
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j.err

# Directories
INDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/rad_reads"
OUTDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping"
REF="/scratch/PNeuhaus/PaulaNeuhaus/Output/mapping/ref_genomes/HM/hypericumpulchrum/hypericumpulchrum_SAMEA10241452.fasta"

# Create output directory if needed
mkdir -p "$OUTDIR"

# Use the random 100 sample list
SAMPLE=$(sed -n "${SLURM_ARRAY_TASK_ID}p" ${INDIR}/sample_list_10_5_new.txt)

FQ1=${INDIR}/${SAMPLE}.1.fq.gz
FQ2=${INDIR}/${SAMPLE}.2.fq.gz
BAM=${OUTDIR}/${SAMPLE}.sorted.bam

echo "Mapping sample: $SAMPLE"
echo "FQ1: $FQ1"
echo "FQ2: $FQ2"
echo "Output BAM: $BAM"

module purge
module load bio/bwa-mem2/2.2.1-intel-compilers-2023.1.0

bwa-mem2 mem -t 6 $REF $FQ1 $FQ2 > ${OUTDIR}/${SAMPLE}.sam


module purge
module load bio/SAMtools/1.19.2-GCC-13.2.0   # Pick a SAMtools version compatible with GCC

samtools view -bS ${OUTDIR}/${SAMPLE}.sam | samtools sort -o $BAM -
samtools index $BAM

rm ${OUTDIR}/${SAMPLE}.sam

4 Deduplication of Mapped Reads

After the mapping, the deduplication was done. Again for every enzyme from 10_5

Script: dedup_radinitio_KpnI.sh

#!/bin/bash -l

#SBATCH --job-name=dedup_radinitio
#SBATCH --nodes=1
#SBATCH --time=2:00:00
#SBATCH --cpus-per-task=6
#SBATCH --mem=24gb
#SBATCH --account=ag-demeaux
#SBATCH --array=1-50
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j_%a.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j_%a.err

module purge
module load bio/SAMtools/1.18-GCC-12.3.0

INLIST="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/rad_reads/sample_list_10_5_new.txt"
INDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping"
OUTDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping/dedup"

mkdir -p $OUTDIR

# Get sample name
SAMPLE=$(sed -n "${SLURM_ARRAY_TASK_ID}p" ${INLIST})

echo "Processing $SAMPLE"

# Input BAM
INBAM=${INDIR}/${SAMPLE}.sorted.bam

# Intermediate + final BAMs
NAME_SORT=${OUTDIR}/${SAMPLE}_name_sorted.bam
FIXMATE=${OUTDIR}/${SAMPLE}_fixmate.bam
FIXMATE_SORT=${OUTDIR}/${SAMPLE}_fixmate_sorted.bam
DEDUP=${OUTDIR}/${SAMPLE}_sorted_dedup.bam
STATS=${OUTDIR}/${SAMPLE}_duplicates_report.txt

# Step 1: sort by name
samtools sort -n -@ 6 -O bam -o $NAME_SORT $INBAM

# Step 2: fixmate
samtools fixmate -m -O bam $NAME_SORT $FIXMATE
rm -f $NAME_SORT

# Step 3: coordinate sort
samtools sort -@ 6 -O bam -o $FIXMATE_SORT $FIXMATE
rm -f $FIXMATE

# Step 4: mark duplicates (remove them with -r, keep stats with -s)
samtools markdup -r -s -@ 6 $FIXMATE_SORT $DEDUP
rm -f $FIXMATE_SORT

# Step 5: stats summary
samtools flagstat $DEDUP > $STATS

samtools index $DEDUP

echo "Finished $SAMPLE"

5 (Multi)BamQC

This was done for every enzyme from 10_5

Script: multibamqc_KpnI.sh

#!/bin/bash -l

#SBATCH --job-name=multibamqc_KpnI
#SBATCH --time=1:00:00
#SBATCH --cpus-per-task=4
#SBATCH --mem=16gb
#SBATCH --account=ag-demeaux
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/logs/%x_%j.err

set -euo pipefail

BAM_DIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping/dedup"
BAMQC_DIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping/dedup/bamqc/"
MULTIBAMQC_DIR="$BAMQC_DIR/multibamqc"

mkdir -p "$BAMQC_DIR" "$MULTIBAMQC_DIR"

# Set Java options for headless mode
export JAVA_OPTS="-Djava.awt.headless=true"

# Run Qualimap BAMQC for each sorted BAM file
for BAM_FILE in "$BAM_DIR"/*_sorted_dedup.bam; do
SAMPLE_NAME=$(basename "$BAM_FILE" _sorted_dedup.bam)
QUALIMAP_OUTPUT_DIR="$BAMQC_DIR/$SAMPLE_NAME"

# Define output and error log files for each sample INSIDE the loop
SBATCH_OUTPUT_FILE="$BAMQC_DIR/${SAMPLE_NAME}_bamqc.out"
SBATCH_ERROR_FILE="$BAMQC_DIR/${SAMPLE_NAME}_bamqc.err"

echo "Running Qualimap BAMQC for $SAMPLE_NAME"
qualimap bamqc \
-bam "$BAM_FILE" \
-outdir "$QUALIMAP_OUTPUT_DIR" \
-outformat HTML \
-outfile "${SAMPLE_NAME}_report" \
-nw 4 \
> "$SBATCH_OUTPUT_FILE" 2> "$SBATCH_ERROR_FILE"
done

# Prepare input groups file for multi-bamqc
INPUT_GROUPS_FILE="$BAMQC_DIR/input_groups.txt"
rm -f "$INPUT_GROUPS_FILE"

echo "Creating input groups file for Multi-BAMQC"
for BAMQC_OUTPUT_DIR in "$BAMQC_DIR"/*; do
if [ -d "$BAMQC_OUTPUT_DIR" ] && [ -f "$BAMQC_OUTPUT_DIR/genome_results.txt" ]; then
SAMPLE_NAME=$(basename "$BAMQC_OUTPUT_DIR")
echo -e "${SAMPLE_NAME}\t${BAMQC_OUTPUT_DIR}\t1" >> "$INPUT_GROUPS_FILE"
fi
done

# Run Qualimap Multi-BAMQC
echo "Running Qualimap Multi-BAMQC"
qualimap multi-bamqc \
-d "$INPUT_GROUPS_FILE" \
-outdir "$MULTIBAMQC_DIR" \
-outformat PDF:HTML \
-nw 4

echo "Qualimap analysis complete. Reports saved in $MULTIBAMQC_DIR."

# List with all the paths to the .sorted.bam files

# Defines Output where to put the vcf

# Print job info to log file
echo "Starting variant calling with bcftools" > $SLURM_JOB_ID.vcf.log
echo "Reference Genome: $REF" >> $SLURM_JOB_ID.vcf.log
echo "BAM list: $BAMLIST" >> $SLURM_JOB_ID.vcf.log
echo "Output VCF: $OUT" >> $SLURM_JOB_ID.vcf.log echo "Start time: $(date)" >> $SLURM_JOB_ID.vcf.log

bcftools mpileup \
-f $REF \
-b $BAMLIST \
-q 20 \
-Q 30 \
-a AD,DP,SP \
-Ou \
| bcftools call \
-m -v \
-a GQ,GP \
-Oz \
-o $OUT

echo "End time: $(date)" >> $SLURM_JOB_ID.vcf.log

OUT="/scratch/PNeuhaus/PaulaNeuhaus/Output/radinitio/hypericumpulchrum/population/10_5/simulation/mapping/dedup/vcf/vcftools"

echo "Starting VCFtools statistics"
echo "VCF: $SET_VCF"
echo "Output directory: $OUT"

# Calculate the allele frequency
vcftools --gzvcf $SET_VCF --freq2 --out $OUT --max-alleles 2

# Calculate the mean depth per individual
vcftools --gzvcf $SET_VCF --depth --out $OUT

# Calculate the mean depth per site
vcftools --gzvcf $SET_VCF --site-mean-depth --out $OUT

# Calculate the site quality
vcftools --gzvcf $SET_VCF --site-quality --out $OUT

# Calculate the proportion of missing data per individual
vcftools --gzvcf $SET_VCF --missing-indv --out $OUT

# Calculate proportion of missing data per site
vcftools --gzvcf $SET_VCF --missing-site --out $OUT

# Calculate the heterozygosity and inbreeding coefficient per individual
vcftools --gzvcf $SET_VCF --het --out $OUT

# Define file paths

# Helper to read tab files with header + 1 header line to skip
read_tab <- function(path, col_names) {
vroom::vroom(path, delim = "\t", skip = 1, col_names = col_names, show_col_types = FALSE)
}

# --- Read files ---
var_qual <- read_tab(file.path(base, "vcftools.lqual"),
c("CHROM","POS","QUAL"))

c("CHROM","POS","MEAN_DEPTH","VAR_DEPTH"))

var_miss <- read_tab(file.path(base, "vcftools.lmiss"),
c("CHR","POS","N_DATA","N_GENOTYPE_FILTERED","N_MISS","F_MISS"))

# --freq2 prints allele:freq; parse the numeric frequencies correctly
