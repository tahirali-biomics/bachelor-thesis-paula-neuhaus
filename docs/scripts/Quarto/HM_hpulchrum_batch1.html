1 Demultiplexing and Cleaning

Script: /scratch/PNeuhaus/PaulaNeuhaus/Scripts/demultiplexing_old_hpu.sh

#!/usr/bin/env bash

#SBATCH --job-name=demux_HM
#SBATCH --time=30:00:00
#SBATCH --cpus-per-task=8
#SBATCH --mem=400G
#SBATCH --account=ag-demeaux
#SBATCH --mail-type=END,FAIL
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/logs/%x_%j.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/logs/%x_%j.err

# ============================================================
# PURPOSE
# ============================================================
# This pipeline:
#
# 1. Trims the 5 bp spacer from R1 only
# 2. Demultiplexes reads using process_radtags
# 3. Optionally removes PCR duplicates AFTER demultiplexing
#
# Library structure identified:
#
#   NNNNN + BARCODE + GGTACC + genomic DNA
#
# where:
#   NNNNN = diversity spacer (5 bp)
#   BARCODE = inline barcode (6 bp)
#   GGTACC = KpnI restriction site
#
# ============================================================

set -euo pipefail
shopt -s nullglob

# ============================================================
# LOAD MODULES
# ============================================================

module purge

module load bio/Stacks/2.62-foss-2022a
module load bio/cutadapt/4.2-GCCcore-11.3.0

# ============================================================
# CONFIGURATION
# ============================================================

THREADS=8

FASTQ_DIR="/scratch/RADseq_ag-deamux/NGS_24D324"
OUTROOT="/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/old"
mkdir -p "$OUTROOT"

BARCODE_FILE="$FASTQ_DIR/barcodes_pool4.txt"

R1="$FASTQ_DIR/DX166_303714_24D324-4_S56_L000_R1_001.fastq.gz"
R2="$FASTQ_DIR/DX166_303714_24D324-4_S56_L000_R2_001.fastq.gz"

ENZYME="kpnI"

# process_radtags options
INLINE_FLAG="--inline_null"
BARCODE_MISMATCH="--barcode-dist-1 1"

# ============================================================
# OUTPUT DIRECTORIES
# ============================================================

LOG_DIR="$OUTROOT/logs"
TRIM_DIR="$OUTROOT/trimmed"
RAD_DIR="$OUTROOT/radtags"
DUP_DIR="$OUTROOT/clone_filtered"

mkdir -p \
    "$LOG_DIR" \
    "$TRIM_DIR" \
    "$RAD_DIR" \
    "$DUP_DIR"

# ============================================================
# CHECK INPUT FILES
# ============================================================

if [[ ! -f "$R1" ]]; then
    echo "ERROR: R1 file not found"
    exit 1
fi

if [[ ! -f "$R2" ]]; then
    echo "ERROR: R2 file not found"
    exit 1
fi

if [[ ! -f "$BARCODE_FILE" ]]; then
    echo "ERROR: Barcode file not found"
    exit 1
fi

echo "=================================================="
echo "JOB STARTED: $(date)"
echo "HOSTNAME: $(hostname)"
echo "=================================================="

# ============================================================
# STEP 1 — TRIM 5 bp SPACER FROM R1 ONLY
# ============================================================
#
# Your data show:
#
#   position 6 = barcode
#   position 12 = GGTACC
#
# meaning:
#
#   5 bp spacer + 6 bp barcode + restriction site
#
# We therefore remove first 5 bp from R1 only.
#
# R2 remains untouched.
#
# ============================================================

echo "=================================================="
echo "STEP 1: TRIMMING R1 SPACER"
echo "=================================================="

TRIM_R1="$TRIM_DIR/trimmed_R1.fastq.gz"
TRIM_R2="$TRIM_DIR/trimmed_R2.fastq.gz"

cutadapt \
    -u 5 \
    -o "$TRIM_R1" \
    -p "$TRIM_R2" \
    "$R1" \
    "$R2" \
    > "$LOG_DIR/cutadapt.log" 2>&1

echo "Spacer trimming completed"

# ============================================================
# STEP 2 — DEMULTIPLEX WITH process_radtags
# ============================================================

echo "=================================================="
echo "STEP 2: process_radtags"
echo "=================================================="

process_radtags \
    -1 "$TRIM_R1" \
    -2 "$TRIM_R2" \
    -o "$RAD_DIR" \
    -b "$BARCODE_FILE" \
    -e "$ENZYME" \
    $INLINE_FLAG \
    $BARCODE_MISMATCH \
    -r -c -q \
    -i gzfastq \
    -T "$THREADS" \
    -D \
    > "$LOG_DIR/process_radtags.log" 2>&1

echo "process_radtags completed"

# ============================================================
# STEP 3 — OPTIONAL PCR DUPLICATE REMOVAL
# ============================================================
#
# This is performed AFTER demultiplexing,
# which is the biologically cleaner workflow.
#
# ============================================================

echo "=================================================="
echo "STEP 3: clone_filter"
echo "=================================================="

DEMUX_R1=("$RAD_DIR"/*.1.fq.gz)

if [[ ${#DEMUX_R1[@]} -eq 0 ]]; then
    echo "ERROR: No demultiplexed files found"
    exit 1
fi

for r1_file in "${DEMUX_R1[@]}"; do

    sample=$(basename "$r1_file" .1.fq.gz)

    r2_file="$RAD_DIR/${sample}.2.fq.gz"

    if [[ ! -f "$r2_file" ]]; then
        echo "WARNING: Missing R2 for sample $sample"
        continue
    fi

    sample_out="$DUP_DIR/$sample"

    mkdir -p "$sample_out"

    echo "Processing clone_filter for sample: $sample"

    clone_filter \
        -1 "$r1_file" \
        -2 "$r2_file" \
        -o "$sample_out" \
        -i gzfastq \
        --oligo-len-1 6 \
        -y gzfastq \
        > "$sample_out/clone_filter.log" 2>&1
done

echo "=================================================="
echo "PIPELINE FINISHED SUCCESSFULLY"
echo "DATE: $(date)"
echo "=================================================="

6 Deduplication of Mapped Reads

dedup_HM.sh

#!/bin/bash -l

#SBATCH --job-name=dedup_hm_old
#SBATCH --nodes=1
#SBATCH --time=2:00:00
#SBATCH --cpus-per-task=6
#SBATCH --mem=24gb
#SBATCH --account=ag-demeaux
#SBATCH --output=/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/logs/%x_%j_%a.out
#SBATCH --error=/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/logs/%x_%j_%a.err
#SBATCH --array=1-51

module purge
module load bio/SAMtools/1.18-GCC-12.3.0

INLIST="/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/old/fastp/sample_list_HM.txt"
INDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/old/mapping"
OUTDIR="/scratch/PNeuhaus/PaulaNeuhaus/Output/duprem/HM/old/mapping/dedup"

mkdir -p $OUTDIR

# Get sample name
SAMPLE=$(sed -n "${SLURM_ARRAY_TASK_ID}p" ${INLIST})

echo "Processing $SAMPLE"

# Input BAM
INBAM=${INDIR}/${SAMPLE}.sorted.bam

# Intermediate + final BAMs
NAME_SORT=${OUTDIR}/${SAMPLE}_name_sorted.bam
FIXMATE=${OUTDIR}/${SAMPLE}_fixmate.bam
FIXMATE_SORT=${OUTDIR}/${SAMPLE}_fixmate_sorted.bam
DEDUP=${OUTDIR}/${SAMPLE}_sorted_dedup.bam
STATS=${OUTDIR}/${SAMPLE}_duplicates_report.txt

# Step 1: sort by name
samtools sort -n -@ 6 -O bam -o $NAME_SORT $INBAM

# Step 2: fixmate
samtools fixmate -m -O bam $NAME_SORT $FIXMATE
rm -f $NAME_SORT

# Step 3: coordinate sort
samtools sort -@ 6 -O bam -o $FIXMATE_SORT $FIXMATE
rm -f $FIXMATE

# Step 4: mark duplicates (remove them with -r, keep stats with -s)
samtools markdup -r -s -@ 6 $FIXMATE_SORT $DEDUP
rm -f $FIXMATE_SORT

# Step 5: stats summary
samtools flagstat $DEDUP > $STATS

samtools index $DEDUP

echo "Finished $SAMPLE"

-f "$REF" \
-b "$BAMLIST" \
-q 20 \
-Q 30 \
-a AD,DP,SP \
-Ou \

-m -v \
-a GQ,GP \
-Oz \
-o "$OUT"

bcftools index "$OUT"

echo "End time: $(date)" >> $SLURM_JOB_ID.vcf.log

summ <- function(x) {
  tibble(
    n      = sum(!is.na(x)),
    min    = min(x, na.rm = TRUE),
    q1     = quantile(x, 0.25, na.rm = TRUE),
    median = median(x, na.rm = TRUE),
    mean   = mean(x, na.rm = TRUE),
    q3     = quantile(x, 0.75, na.rm = TRUE),
    max    = max(x, na.rm = TRUE),

  )
}
